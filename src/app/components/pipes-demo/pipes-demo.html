Let's Learn Angular Pipes

uppercase, lowercase, titlecase pipe:
myName without any pipe: {{myName}}
myName with uppercase any pipe: {{myName | uppercase}}
myName with lowercase any pipe: {{myName | lowercase}}
myName with titlecase any pipe: {{myName | titlecase}}
currency pipe:
mySal without any pipe: {{mySal}}
mySal with currency pipe: {{mySal | currency}}
mySal with currency pipe: {{mySal | currency : 'USD' : 'symbol'}}
mySal with currency pipe: {{mySal | currency : 'USD' : 'code'}}
mySal with currency pipe: {{mySal | currency : 'INR' : 'symbol'}}
mySal with currency pipe: {{mySal | currency : '€'}}
mySal with currency pipe: {{mySal | currency : '元'}}
